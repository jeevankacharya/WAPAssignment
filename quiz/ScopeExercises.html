10. Write a Javascript function that takes any number of input arguments and returns the product of the arguments.

Ans:


                                 function multiply(){

                                       var result=1;

                                       for(int i=0;i<arguments.length;i++){

                                              result = result * arguments[i];

                                           }

                                         return result;

                                }
