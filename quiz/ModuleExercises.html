12. Rewrite your answer to Question 10 using the Locally Scoped Object Literal Pattern.

Ans :

 var name,age,salary;

  }

  }

    return this.salary;

  }

             this.age = newage;

         }


  employeeObject.setSalary= function(newSalary){

          this.salary = newSalary;

       }


  employeeObject.setName = function(newname){

          this.name = newname;

         }

        }
